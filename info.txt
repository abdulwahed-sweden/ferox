## 🔧 Smart Build System

### Intelligent Build Script
```rust
// build.rs
use std::{collections::HashMap, env, fs, path::Path};

fn main() -> Result<(), Box<dyn std::error::Error>> {
    println!("cargo:rerun-if-changed=src");
    println!("cargo:rerun-if-changed=Cargo.toml");
    
    let manifest = ModuleManifest::load()?;
    let mut report = BuildReport::new();
    
    // Validate module structure
    for (category, modules) in &manifest.categories {
        for module in modules {
            let path = format!("src/{}/{}.rs", category, module);
            if Path::new(&path).exists() {
                report.add_success(category, module);
                println!("cargo:warning=✅ {}::{} registered", category, module);
            } else {
                report.add_missing(category, module);
                println!("cargo:warning=❌ {}::{} - FILE MISSING", category, module);
            }
        }
    }
    
    // Generate module registry
    if let Err(e) = generate_module_registry(&manifest) {
        eprintln!("cargo:warning=Failed to generate module registry: {}", e);
    }
    
    Ok(())
}
```

### Automatic Registry Generator
```rust
// src/tools/registry_generator.rs
pub struct RegistryGenerator {
    manifest: ModuleManifest,
}

impl RegistryGenerator {
    pub fn generate_rust_code(&self) -> String {
        let mut code = String::from("// Auto-generated module registry\n\n");
        
        for (category, data) in &self.manifest.categories {
            code.push_str(&format!("pub mod {} {{\n", category));
            
            for module in &data.modules {
                code.push_str(&format!("    pub mod {};\n", module));
            }
            
            code.push_str("}\n\n");
        }
        
        code
    }
    
    pub fn write_registry_file(&self) -> Result<(), std::io::Error> {
        let code = self.generate_rust_code();
        std::fs::write("src/modules/auto_generated.rs", code)
    }
}
```

use std::path::Path;

#[test]
fn test_memory_forensics_modules_exist() {
    let expected_modules = [
        "dump_parser",
        "process_analyzer", 
        "malware_detector",
        "network_analyzer",
        "registry_analyzer",
        "credential_extractor",
        "mitre_mapper",
        "volatility_bridge"
    ];
    
    for module in expected_modules {
        let path = format!("src/memory_forensics/{}.rs", module);
        assert!(
            Path::new(&path).exists(),
            "Missing memory forensics module: {}",
            module
        );
    }
}

#[test]
fn test_module_registration() {
    // Test that all modules are properly registered
    let manifest = ferox_tools::manifest::load_manifest().unwrap();
    
    for (category, data) in manifest.categories {
        for module in data.modules {
            // Test Rust module declaration exists
            let mod_path = format!("src/{}/mod.rs", category);
            let mod_content = std::fs::read_to_string(mod_path).unwrap();
            
            assert!(
                mod_content.contains(&format!("pub mod {};", module)),
                "Module {} not registered in {}/mod.rs",
                module,
                category
            );
        }
    }
}
